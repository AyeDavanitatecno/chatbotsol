        <button onclick="sendMessage()">Enviar</button>

    </div>

    <script>

        function sendMessage() {

            let input = document.getElementById("user-input").value;

            if (input.trim() === "") return;

           

            let chatBox = document.getElementById("chat-box");

            let userMessage = `<p><strong>Tú:</strong> ${input}</p>`;

            chatBox.innerHTML += userMessage;

            document.getElementById("user-input").value = "";

           

            setTimeout(() => {

                let response = getResponse(input);

                let botMessage = `<p><strong>Sol:</strong> ${response}</p>`;

                chatBox.innerHTML += botMessage;

                chatBox.scrollTop = chatBox.scrollHeight;

            }, 1000);

        }

        function getResponse(input) {

            let responses = {

                "hola": "¡Hola, pequeño planeta! ¿En qué puedo iluminarte hoy? ??",

                "cuéntame sobre marte": "Marte es mi hijo rojo y temperamental. Tiene tormentas de polvo gigantes y sueña con albergar vida algún día. ??",

                "cuéntame sobre júpiter": "Júpiter es el gigante de la familia. Su gran mancha roja es una tormenta que ha durado siglos. ¡Menudo carácter! ???"

            };

            return responses[input.toLowerCase()] || "Lo siento, no entiendo tu pregunta. Pregunta sobre un planeta o salúdame ??";

        }

    </script>
